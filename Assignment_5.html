4.find optimal number of clusters---elbow method¶

In [24]:
from sklearn.cluster import KMeans
wcss=[]
for k in range(1,11):
    kmeans=KMeans(n_clusters=k,random_state=42)
    kmeans.fit(features)
    wcss.append(kmeans.inertia_)
