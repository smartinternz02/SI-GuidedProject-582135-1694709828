  1. Outlier Detection
In[36]:
# Calculate the IQR for Age columns
Q1_age = data["Age"].quantile(0.25)
Q3_age = data["Age"].quantile(0.75)
IQR_age = Q3_age - Q1_age

# Define the lower and upper bounds for non-outliers
lower_bound_age = Q1_age - 1.5 * IQR_age
upper_bound_age = Q3_age + 1.5 * IQR_age

# Replace outliers in the Age column with the nearest non-outlier value
data["Age"] = data["Age"].apply(lambda x: lower_bound_age if x < lower_bound_age else (upper_bound_age if x > upper_bound_age else x))

# Calculate the IQR for Fare columns
Q1_fare = data["Fare"].quantile(0.25)
Q3_fare = data["Fare"].quantile(0.75)
IQR_fare = Q3_fare - Q1_fare

#define the lower and upper bounds for non-outliers
lower_bound_fare = Q1_fare - 1.5 * IQR_fare
upper_bound_fare = Q3_fare + 1.5 * IQR_fare

#Replace outliers in the Age column with the nearest non-outlier value
data["Fare"] = data["Fare"].apply(lambda x: lower_bound_fare if x < lower_bound_fare else (upper_bound_fare if x > upper_bound_fare else x))
